\begin{tikzpicture}
  % Blocs
  \node[block={2.5cm}{2cm}] (P) {P};

      \begin{pmatrix}
        K_{T_x} & 0      & \cdots & 0            \\

  \coordinate[] (inputu)  at ($(P.south west)!0.25!(P.north west)$);
